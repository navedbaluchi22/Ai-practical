from sklearn.neural_network import MLPClassifier

# Training data
X = [[1], [2], [3], [4], [5], [6], [7], [8]]
y = [0, 0, 0, 0, 1, 1, 1, 1]

# Create and train the model
model = MLPClassifier(hidden_layer_sizes=(5,), max_iter=1000, random_state=42)
model.fit(X, y)

# Test data
test_hours = [[10], [3], [6], [8]]
predictions = model.predict(test_hours)

# Display results
print("Hours studied:", test_hours)
print("Predicted result (0=Fail, 1=Pass):", predictions)
